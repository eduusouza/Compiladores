void execSemP(int PID){
    output(PID);
}

void main(void){
    int opcao;
    int PID;
    int qntProcessos;

    while(1 > 0){
        LCDWrite(1);
        opcao = input();

        if(opcao == 1){
            LCDWrite(3);
            PID = input();
            execSemP(PID);          
        } else if(opcao == 2){
            LCDWrite(2);
            qntProcessos = input();
        }
    }
}
