/* Programa para o calculo do mmc entre dois numeros */

int resto (int a, int b){
    int c;
    c = a /b;
    return (a-b*c);
}

int mdc (int a, int b){
    int r;
    while (b != 0){
        r = resto(a , b);
        a = b;
        b = r;
    }
    return a;
}

int mmc (int a, int b){
    int c;
    int temp;
    c = (a*b);
    temp = mdc(a,b);
    c = c/temp;
    return c;
}

void main (void){
    int numUM;
    int numDOIS;
    int resultado;
    numUM = input();
    numDOIS = input ();
    resultado = mmc(numUM, numDOIS); 
    output(resultado);
}
