/* Programa para o calculo da media de 5 numeros */

int vet[5];

int media (int v [] , int tam){
    int soma;
    int aux;
    aux = 0;
    soma = 0;
    while (aux < tam){
        soma = soma + v[aux];
        aux = aux + 1;
    }
    return soma / tam ;
}

void main (void) {
    int i;
    i = 0;
    while (i < 5){
		vet[i] = input();
		i = i + 1;
	}
    output (media (vet , 5));
}
